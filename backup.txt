CREATE TABLE Personas (
    idpersona SERIAL PRIMARY KEY,
    nombres VARCHAR(50) NOT NULL,
    apellidos VARCHAR(60) NOT NULL,
    ci VARCHAR(20) NOT NULL,
    fechareg DATE NOT NULL,
    estado VARCHAR(20) NOT NULL
);
CREATE TABLE Roles (
idrol SERIAL PRIMARY KEY,
nombre VARCHAR(25) NOT NULL,
fechareg DATE NOT NULL,
estado VARCHAR(20) NOT NULL
);

CREATE TABLE Usuarios (
    idusuario SERIAL PRIMARY KEY,
    idpersona INT NULL,
    idrol INT NOT NULL,
    username VARCHAR(50) UNIQUE NOT NULL,
    password_hash TEXT NOT NULL,
    fechareg DATE NOT NULL,
    estado VARCHAR(20) NOT NULL,
    CONSTRAINT fk_usuario_persona FOREIGN KEY (idpersona) REFERENCES Personas(idpersona),
    CONSTRAINT fk_usuario_rol FOREIGN KEY (idrol) REFERENCES Roles(idrol)
);

CREATE TABLE Permisos (
idpermiso SERIAL PRIMARY KEY,
nombre VARCHAR(25) NOT NULL
);

CREATE TABLE rol_permiso (
idrolpermiso SERIAL PRIMARY KEY,
idrol INT NOT NULL,
idpermiso INT NOT NULL,
estado VARCHAR(15) NOT NULL,
CONSTRAINT fk_rol_rolpermiso FOREIGN KEY (idrol) REFERENCES Roles(idrol),
CONSTRAINT fk_permiso_rolpermiso FOREIGN KEY (idpermiso) REFERENCES Permisos(idpermiso)
);
